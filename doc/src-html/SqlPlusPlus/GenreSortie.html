001    package SqlPlusPlus;
002    
003    /** Type enuméré pour gérer les types de sortie possibles
004     * 
005     * @author Dej
006     *
007     */
008    public  enum GenreSortie { 
009            /** Sortie mode TEXTE */
010            TXT("TEXTE"), 
011            /** Sortie mode HTML */
012            HTML("HTML"),
013            /** Sortie mode XML */
014            XML("XML"),
015            /** Sortie mode CSV */
016            CSV("CSV");
017    
018            /** Libelle associé au genre */
019            private String libelle;
020    
021            /** Accès au libellé */
022            public String getLabel(){ return libelle; }
023    
024            /** Construction de la liste des valeurs du type enuméré
025             * @param simple <code>true</code> ("TEXTE",...), <code>false</code> (TXT=>"TEXTE",...)
026             * @return La chaîne créée
027             *  */
028            public static String liste(boolean simple){
029                    StringBuffer s=new StringBuffer();
030                    String sep = "";
031                    s.append("(");
032                    for (GenreSortie g : GenreSortie.values())
033                    {
034                            s.append(sep);
035                            if (!simple) s.append(g.toString()+"=>");
036                            s.append("\""+ g.libelle+"\"");
037                            sep=",";
038                    }
039                    s.append(")");
040    
041                    return s.toString();
042            }
043            /**  Construction de la liste version longue des valeurs du type enuméré (liste(true) càd (TXT=>"TEXTE",...) )
044             * @return La chaîne créée
045             * */ 
046            public static String liste() { return liste(false); }
047            
048            /** Accès au type énuméré à partir de son libellé
049             * 
050             * @param qui Le libellé recherché
051             * @return Le type (<code>null</code> si pas trouvé)
052             */
053            public static GenreSortie lequel(String qui)
054            {
055                    for (GenreSortie e : GenreSortie.values() )
056                            if (qui.equalsIgnoreCase(e.libelle))
057                                    return e;
058                    return null;
059            }
060            
061            /** Constructeur avec le libellé
062             * @param s Le libellé
063             */
064            GenreSortie(String s){libelle=s;}
065    }
